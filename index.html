☁️ اتصال به Google Sheets

برای اشتراک‌گذاری داده بین چند کاربر، یک Google Apps Script روی یک Sheet تنظیم کنید. فقط یک بار این کار رو انجام بدید:

۱ یک Google Sheet جدید باز کنید ← از منو Extensions → Apps Script را انتخاب کنید
۲ کد زیر را کامل جایگزین کد پیش‌فرض کنید: 
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Dashboard') || ss.insertSheet('Dashboard');
  const val = sheet.getRange('A1').getValue();
  const data = val || '{"version":1,"MODELS":[],"INVENTORY":[],"ORDERS":[],"USER_CHARTS":[],"settings":{},"THEME":{},"savedAt":null}';
  return ContentService.createTextOutput(data).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName('Dashboard') || ss.insertSheet('Dashboard');
    sheet.getRange('A1').setValue(JSON.stringify(body.data));
    return ContentService.createTextOutput('{"ok":true}').setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput('{"ok":false,"error":"'+err.message+'"}').setMimeType(ContentService.MimeType.JSON);
  }
}
۳ Deploy → New deployment ← نوع: Web app ← Execute as: Me ← Access: Anyone ← آدرس URL را کپی کنید
این آدرس را فقط برای کسانی بفرستید که می‌خواهید داده را ببینند یا ویرایش کنند
